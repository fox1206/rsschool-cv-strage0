Task from Codewars:

Given a string s. You have to return another string such that even-indexed and odd-indexed characters of s are grouped and groups are space-separated.

function sortMyString(S) {
    let text = '';
    let even = '';
    let odd = '';
    for(let i = 0; i < S.length; i++){
        if(i % 2 == 0){
            even += S[i];
        }
        else{
            odd += S[i];
        }
    }
    text = even + ' ' + odd;
    return text;
}
